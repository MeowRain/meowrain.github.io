```javascript
//String to Int
let num = '10'
//第一种
let numInt = parseInt(num)
console.log(numInt);

//第二种
let numInt2 = Number(num);
console.log(numInt2);

//第三种
let numInt3 = +num;
console.log(numInt3);

//String to Float
let num_1 = '10.5'
//第一种
let numFloat = parseFloat(num_1);
console.log(numFloat);
//第二种
let numFloat_1 = Number(num_1);
console.log(numFloat_1);
//第三种
let numFloat_2 = +num_1;
console.log(numFloat_2);


//Float to Int
let num_2 = 9.81
let numInt4 = parseInt(num_2);
console.log(numInt4);


//Int to String
let num_3 = 10
let numString = num_3.toString();
console.log(numString);

//Float to String
let num_4 = 10.5
let numString_1 = num_4.toString();
console.log(numString_1);

//Int to Float
let num_5 = 10
let numFloat_3 = parseFloat(num_5);
console.log(numFloat_3);

```
